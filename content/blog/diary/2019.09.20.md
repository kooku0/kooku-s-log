* 알고리즘 문제 풀기

  * <https://www.acmicpc.net/problem/11729>
  * <https://www.acmicpc.net/problem/10815>
  * <https://www.acmicpc.net/problem/1260>

  ```c++
  #include <iostream>
  #include <vector>
  #include <queue>
  #include <algorithm>
  
  using namespace std;
  
  vector<vector<int>> v;
  vector<bool> visited;
  
  void DFS(int root) {
  	visited[root] = true;
  	cout << root << " ";
  	for (int i = 0; i < v[root].size(); i++) {
  		int dst = v[root][i];
  		if (!visited[dst]) {
  			DFS(dst);
  		}
  	}
  }
  
  void BFS(int root) {
  	visited[root] = true;
  	queue<int> q;
  	q.push(root);
  	cout << root << " ";
  	while (!q.empty()) {
  		int qSize = q.size();
  		for (int i = 0; i < qSize; i++) {
  			int front = q.front();
  			q.pop();
  			for (int i = 0; i < v[front].size(); i++) {
  				int dst = v[front][i];
  				if (!visited[dst]) {
  					visited[dst] = true;
  					q.push(dst);
  					cout << dst << " ";
  				}
  			}
  		}
  	}
  }
  
  
  int main() {
  	ios_base::sync_with_stdio(false);
  	cin.tie(NULL);
  	cout.tie(NULL);
  
  	int N, M, V;
  	cin >> N >> M >> V;
  
  	v.assign(N+1, vector<int>(0));
  	visited.assign(N + 1,false);
  	for (int i = 0; i < M; i++) {
  		int edge, dest;
  		cin >> edge >> dest;
  		v[edge].push_back(dest);
  		v[dest].push_back(edge);
  	}
  	for (auto i : v) {
  		sort(i.begin(), i.end());
  	}
  	DFS(V);
  	visited.assign(N + 1, false);
  	cout << endl;
  	BFS(V);
  	visited.assign(N + 1, false);
  	
  	return 0;
  }
  
  ```
